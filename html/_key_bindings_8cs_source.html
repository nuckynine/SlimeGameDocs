50 private void Start()
51 {
52 foreach (var action in actionAsset)
53 {
54 string actionName = action.name;
55 LoadUpdatedBindings(action);
56 if (action.bindings.Count < 1) { continue; }
57 if (action.bindings[0].isComposite)
58 {
59 Dictionary<string, List<int>> bindingsDict = new Dictionary<string, List<int>>();
60 // start from index 1, since index 0 is the composite
61 for (int i = 1; i < action.bindings.Count; i++)
62 {
63 if (action.bindings[i].isPartOfComposite)
64 {
65 if (bindingsDict.ContainsKey(action.bindings[i].name))
66 {
67 bindingsDict[action.bindings[i].name].Add(i);
68 }
69 else
70 {
71 bindingsDict.Add(action.bindings[i].name, new List<int>());
72 bindingsDict[action.bindings[i].name].Add(i);
73 }
74 }
75 }
76 foreach (var entry in bindingsDict)
77 {
78 InstantiateActionAsset(action, actionName, entry.Value[0], entry.Value[1], true, false, entry.Key);
79 }
80 }
81 else
82 {
83 if (actionName == "Look") { continue; } // don't allow rebinding of look, at least for now
84 string primary = "None";
85 if (action.bindings.Count > 0)
86 primary = InputControlPath.ToHumanReadableString(action.bindings[0].effectivePath);
87 string secondary = "None";
88 if (action.bindings.Count > 1)
89 secondary = InputControlPath.ToHumanReadableString(action.bindings[1].effectivePath);
90
91 InstantiateActionAsset(action, actionName, 0, 1, false, false);
92 }
93 }

115 private static void DoRebind(InputAction actionToRebind, int bindingIndex, TMP_Text statusText, bool allCompositeParts, bool excludeMouse)
116 {
117 if (actionToRebind == null || bindingIndex < 0) { return; }
118
119 statusText.text = "Enter Control";
120 actionToRebind.Disable();
121
122 var rebind = actionToRebind.PerformInteractiveRebinding(bindingIndex);
123 rebind.OnComplete(operation =>
124 {
125 actionToRebind.Enable();
126 operation.Dispose();
127
128 if (allCompositeParts)
129 {
130 var nextBindingIndex = bindingIndex + 1;
131 if (nextBindingIndex < actionToRebind.bindings.Count && actionToRebind.bindings[nextBindingIndex].isComposite)
132 DoRebind(actionToRebind, nextBindingIndex, statusText, true, excludeMouse);
133 }
134
135 SaveBindingAction(actionToRebind);
136 rebindComplete?.Invoke();
137 });
138
139 rebind.OnCancel(operation =>
140 {
141 actionToRebind.Enable();
142 operation.Dispose();
143 rebindCanceled?.Invoke();
144 });
145
146 rebind.WithCancelingThrough("<Keyboard>/escape");
147
148 if (excludeMouse)
149 rebind.WithControlsExcluding("Mouse");
150
151 rebindStarted?.Invoke(actionToRebind, bindingIndex);
152 rebind.Start();
153 }

171 public static void LoadUpdatedBindings(InputAction action)
172 {
173 for (int i = 0; i < action.bindings.Count; i++)
174 {
175 if (!string.IsNullOrEmpty(PlayerPrefs.GetString(action.actionMap + action.name + i)))

181 public static void ResetBinding(InputAction action, int bindingIndex)
182 {
183 if (action == null || action.bindings.Count <= bindingIndex)
184 {
185 Debug.Log("Could not find action or binding");
186 return;
187 }
188 if (action.bindings[bindingIndex].isComposite)
189 {
190 for (int i = bindingIndex; i < action.bindings.Count && action.bindings[i].isComposite; i++)
191 action.RemoveBindingOverride(i);
192 }
193 }
