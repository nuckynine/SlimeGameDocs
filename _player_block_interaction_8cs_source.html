162 private void RaycastToEdgeBlock()
163 {
164 var camTrans = _mainCamera.transform;
165 var camPos = camTrans.position;
166 var camForward = camTrans.forward;
167
168 Plane groundPlane = new Plane(Vector3.up, _groundPlane.position);
169 Ray lookRay = new Ray(camPos, camForward);
170 groundPlane.Raycast(lookRay, out float groundPlaneHitDist);
171
172 // Exit early if beyond max interaction distance
173 if (groundPlaneHitDist > _interactionDistance)
174 {
175 Indicator.transform.position = _indicatorHidePos;
176 return;
177 }
178
179 Vector3 checkPos = lookRay.GetPoint(groundPlaneHitDist);
180
181 if (Physics.Raycast(checkPos, Vector3.forward, out _adjacentHit, _EdgeCheckRange, _placementLayer)){}
182 else if (Physics.Raycast(checkPos, Vector3.back, out _adjacentHit, _EdgeCheckRange, _placementLayer)){}
183 else if (Physics.Raycast(checkPos, Vector3.left, out _adjacentHit, _EdgeCheckRange, _placementLayer)){}
184 else if (Physics.Raycast(checkPos, Vector3.right, out _adjacentHit, _EdgeCheckRange, _placementLayer)){}
185 else
186 {
187 Indicator.transform.position = _indicatorHidePos;
188 }
