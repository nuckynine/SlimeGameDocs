290 private void PickUpOrPlace(ContainerSlot source)
291 {
292 //this determines if we should pick up an item onto the cursor or drop it onto a slot
293 if (IsNullOrEmpty(cursorSlot.GetItemStack()))
294 {
295 //if cursor has no item
296 if (!IsNullOrEmpty(source.GetItemStack()))
297 {
298 //if slot has item, pick it up
299 ItemStack moved = source.RemoveItemStack();
300 cursorSlot.AddNewItemStack(moved);
301 }
302 }
303 else
304 {
305 //if cursor has item
306 if (IsNullOrEmpty(source.GetItemStack()) || source.GetItemStack().GetItemName() == cursorSlot.GetItemStack().GetItemName())
307 {
308 //if slot under cursor has no item/item can be merged, put from cursor into slot
309 ItemStack moved = cursorSlot.RemoveItemStack();
310 int newSize = moved.GetStackSize() + (source.GetItemStack() == null ? 0 : source.GetItemStack().GetStackSize());
311 moved.SetStackSize(newSize);
312 source.AddNewItemStack(moved);
313 }
314 else
315 {
316 //this means that there is 2 different items, swap their places
317 ItemStack moved = cursorSlot.RemoveItemStack();
318 cursorSlot.AddNewItemStack(source.GetItemStack());
319 source.AddNewItemStack(moved);
320 }
321 }

323 private void SplitAndPickUpOrPlace(ContainerSlot source)
324 {
325 if (IsNullOrEmpty(cursorSlot.GetItemStack()))
326 {
327 if (!IsNullOrEmpty(source.GetItemStack()))
328 {
329 //if no item in cursor, but item in slot grab half of the stack with the cursor
330 ItemStack moved = source.GetItemStack().Clone();
331 int amount = moved.GetStackSize() / 2;
332 source.GetItemStack().SetStackSize(amount);
333 moved.SetStackSize(moved.GetStackSize() - amount);
334 cursorSlot.AddNewItemStack(moved);
335 }
336 }
337 else
338 {
339 if (IsNullOrEmpty(source.GetItemStack()) || source.GetItemStack().GetItemName() == cursorSlot.GetItemStack().GetItemName())
340 {
341
342 int amount = cursorSlot.GetItemStack().GetStackSize();
343 amount = amount == 1 ? amount : amount / 2;
344 cursorSlot.GetItemStack().RemoveFromStack(amount);
345 if (!IsNullOrEmpty(source.GetItemStack()))
346 {
347 source.GetItemStack().AddToStack(amount);
