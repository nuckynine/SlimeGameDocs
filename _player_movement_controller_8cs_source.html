246 private void Move()
247 {
248 // set target speed based on move speed, sprint speed and if sprint is pressed
249 float targetSpeed = _input.sprint ? SprintSpeed : MoveSpeed;
250
251 // a simplistic acceleration and deceleration designed to be easy to remove, replace, or iterate upon
252
253 // note: Vector2's == operator uses approximation so is not floating point error prone, and is cheaper than magnitude
254 // if there is no input, set the target speed to 0
255 if (_input.move == Vector2.zero) targetSpeed = 0.0f;
256
257 // a reference to the players current horizontal velocity
258 float currentHorizontalSpeed = new Vector3(_controller.velocity.x, 0.0f, _controller.velocity.z).magnitude;
259
260 float speedOffset = 0.1f;
261 float inputMagnitude = _input.analogMovement ? _input.move.magnitude : 1f;
262
263 // accelerate or decelerate to target speed
264 if (currentHorizontalSpeed < targetSpeed - speedOffset || currentHorizontalSpeed > targetSpeed + speedOffset)
265 {
266 // creates curved result rather than a linear one giving a more organic speed change
267 // note T in Lerp is clamped, so we don't need to clamp our speed
268 _speed = Mathf.Lerp(currentHorizontalSpeed, targetSpeed * inputMagnitude, Time.deltaTime * SpeedChangeRate);
269
270 // round speed to 3 decimal places
271 _speed = Mathf.Round(_speed * 1000f) / 1000f;
272 }
273 else
274 {
275 _speed = targetSpeed;
276 }
277 _animationBlend = Mathf.Lerp(_animationBlend, targetSpeed, Time.deltaTime * SpeedChangeRate);
278
279 // normalise input direction
280 Vector3 inputDirection = new Vector3(_input.move.x, 0.0f, _input.move.y).normalized;
281
282 // note: Vector2's != operator uses approximation so is not floating point error prone, and is cheaper than magnitude
283 // if there is a move input rotate player when the player is moving
284 if (_input.move != Vector2.zero)
285 {
286 _targetRotation = Mathf.Atan2(inputDirection.x, inputDirection.z) * Mathf.Rad2Deg + _mainCamera.transform.eulerAngles.y;
287 float rotation = Mathf.SmoothDampAngle(transform.eulerAngles.y, _targetRotation, ref _rotationVelocity, RotationSmoothTime);
288
289 // rotate to face input direction relative to camera position
290 transform.rotation = Quaternion.Euler(0.0f, rotation, 0.0f);
291 }
292
293
294 Vector3 targetDirection = Quaternion.Euler(0.0f, _targetRotation, 0.0f) * Vector3.forward;
295
296 // move the player
297 _controller.Move(targetDirection.normalized * (_speed * Time.deltaTime) + new Vector3(0.0f, _verticalVelocity, 0.0f) * Time.deltaTime);
298
299 // update animator if using character
300 if (_hasAnimator)
301 {
302 _animator.SetFloat(_animIDSpeed, _animationBlend);
303 _animator.SetFloat(_animIDMotionSpeed, inputMagnitude);
304 }
305 }

307 private void JumpAndGravity()
308 {
309 if (Grounded)
310 {
311 // reset the fall timeout timer
312 _fallTimeoutDelta = FallTimeout;
313
314 // update animator if using character
315 if (_hasAnimator)
316 {
317 _animator.SetBool(_animIDJump, false);
318 _animator.SetBool(_animIDFreeFall, false);
319 }
320
321 // stop our velocity dropping infinitely when grounded
322 if (_verticalVelocity < 0.0f)
323 {
324 _verticalVelocity = -2f;
325 }
326
327 // Jump
328 if (_input.jump && _jumpTimeoutDelta <= 0.0f)
329 {
330 // the square root of H * -2 * G = how much velocity needed to reach desired height
331 _verticalVelocity = Mathf.Sqrt(JumpHeight * -2f * Gravity);
332
333 // update animator if using character
334 if (_hasAnimator)
335 {
336 _animator.SetBool(_animIDJump, true);
337 }
338 }
339
340 // jump timeout
341 if (_jumpTimeoutDelta >= 0.0f)
342 {
343 _jumpTimeoutDelta -= Time.deltaTime;
344 }
345 }
346 else
347 {
348 // reset the jump timeout timer
349 _jumpTimeoutDelta = JumpTimeout;
350
351 // fall timeout
352 if (_fallTimeoutDelta >= 0.0f)
353 {
354 _fallTimeoutDelta -= Time.deltaTime;
355 }
356 else
357 {
358 // update animator if using character
359 if (_hasAnimator)
360 {
361 _animator.SetBool(_animIDFreeFall, true);
362 }
363 }
364
365 // if we are not grounded, do not jump
366 _input.jump = false;
367 }
368
369 // apply gravity over time if under terminal (multiply by delta time twice to linearly speed up over time)
370 if (_verticalVelocity < _terminalVelocity)
371 {
372 _verticalVelocity += Gravity * Time.deltaTime;
373 }
374 }
